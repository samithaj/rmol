191  (const double iMean, const double iSD,

201  //STDAIR_LOG_DEBUG ("e: " << e);

203  //STDAIR_LOG_DEBUG ("e: " << e);

211  }
212  d1 = 0.5 * (1 - s);
213  }

220  e = exp (e);

222  //STDAIR_LOG_DEBUG ("d2: " << d2);

226  }
